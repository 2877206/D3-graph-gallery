Building shapes in d3.js




The d3.js allows to draw shapes, that together build a graph. This document describe a few helpers function allowing to draw svg from data more efficiently.

area

Allows to draw an area. Works like that:.

// create svg element
var svg = d3.select("#area")
  .append("svg")
    .attr("width", 1000)

// build data
var data = { d: [ [100,300,500], [10,100,600]] }

// add a area
svg
  .selectAll("toto")
  .data(data)
  .enter()
  .append("path")
  .attr("d", function(d){console.log(d)})

Note that this is exactly the same as writing this in your html file:

<area cx="300" cy="70" r="60" fill="#69b3a2" stroke="black" stroke-width="5"></area>

pie

Allows to draw an pie. Works like that:.

// create svg element
var svg = d3.select("#pie")
  .append("svg")
    .attr("width", 1000)
    .attr("height", 400)

// add an pie
svg
  .append("path")
  .attr("transform", "translate(400,200)")
  .attr("d", d3.ribbon()     // imagine your doing a part of a donut plot
    .radius(2)
    )

Note that this is exactly the same as writing this in your html file. This is why helpers function of d3 are very handy

arc

Allows to draw an arc. Works like that:.

// create svg element
var svg = d3.select("#arc")
  .append("svg")
    .attr("width", 1000)
    .attr("height", 400)

// add an arc
svg
  .append("path")
  .attr("transform", "translate(400,200)")
  .attr("d", d3.arc()     // imagine your doing a part of a donut plot
    .innerRadius( 100 )
    .outerRadius( 150 )
    .startAngle( 3.14 )
    .endAngle( 6.28 )
    )

Note that this is exactly the same as writing this in your html file. This is why helpers function of d3 are very handy
