This is a very basic heatmap made using d3.js. The data comes from here.

← Edit me! 
<!DOCTYPE html>
<meta charset="utf-8">

<!-- Load d3.js -->
<script src="http://d3js.org/d3.v4.js"></script>

<!-- Load the color brewer color palette -->
<script src="LIB/colorbrewer.min.js"></script>

<!-- Create a div where the graph will take place -->
<div id="my_dataviz"></div>
← Edit me! 
<script>

      // Feature of the SVG area:
      var margin = { top: 50, right: 0, bottom: 100, left: 130 },
          width = 960 - margin.left - margin.right,
          height = 430 - margin.top - margin.bottom

      // create svg area
      var svg = d3.select("#my_dataviz").append("svg")
          .attr("width", width + margin.left + margin.right)
          .attr("height", height + margin.top + margin.bottom)
          .append("g")
          .attr("transform", "translate(" + margin.left + "," + margin.top + ")");

      // Related to heatmap
      var gridSize = Math.floor(width / 24) // 24 is the number of square per line. gridSize is the size of a square in px
          buckets = 9,
          colors = colorbrewer.Blues[9],  // This gets 9 colors from the Blues color palette of ColorBrewer
          days = ["Mo", "Tu", "We", "Th", "Fr", "Sa", "Su"],
          times = ["1a", "2a", "3a", "4a", "5a", "6a", "7a", "8a", "9a", "10a", "11a", "12a", "1p", "2p", "3p", "4p", "5p", "6p", "7p", "8p", "9p", "10p", "11p", "12p"];


      // Y axis
      var dayLabels = svg.selectAll(".dayLabel")
          .data(days)
          .enter().append("text")
            .text(function (d) { return d; })
            .attr("x", 0)
            .attr("y", function (d, i) { return i * gridSize; })
            .style("text-anchor", "end")
            .attr("transform", "translate(-6," + gridSize / 1.5 + ")")

      // X axis
      var timeLabels = svg.selectAll(".timeLabel")
          .data(times)
          .enter().append("text")
            .text(function(d) { return d; })
            .attr("x", function(d, i) { return i * gridSize; })
            .attr("y", 0)
            .style("text-anchor", "middle")
            .attr("transform", "translate(" + gridSize / 2 + ", -6)")

      // add heatmap cells
      var heatmapChart = d3.tsv("https://raw.githubusercontent.com/holtzy/D3-graph-gallery/master/DATA/data_simple_heatmap.tsv",
        function(d) {
          return {
            day: +d.day,
            hour: +d.hour,
            value: +d.value
          };
        },
        function(data) {
          var colorScale = d3.scaleQuantile()
              .domain([0, buckets - 1, d3.max(data, function (d) { return d.value; })])
              .range(colors);

          svg.selectAll()
              .data(data, function(d) {return d.day+':'+d.hour;})
              .enter()
              .append("rect")
              .attr("x", function(d) { return (d.hour - 1) * gridSize; })
              .attr("y", function(d) { return (d.day - 1) * gridSize; })
              .attr("width", gridSize)
              .attr("height", gridSize)
              .style("fill", function(d) { return colorScale(d.value); });
        });

</script>
